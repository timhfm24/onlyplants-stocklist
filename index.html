function plantListItem(p) { let type = p.category; const typeLabel = type.charAt(0).toUpperCase() + type.slice(1); const status = p.state || 'verfügbar'; // PREIS AUS DEM OBJEKT EINLESEN UND ANZEIGEN const price = (typeof p.preis !== 'undefined' && p.preis !== null) ? `${Number(p.preis).toFixed(2)} €` : ''; // Tag-Klasse je nach Status let tagClass = 'ok'; if(status === 'reserviert') tagClass = 'warn'; if(status === 'nicht verfügbar') tagClass = 'sold'; return `
  • ${escapeHTML(p.name)}
    ${typeLabel}
    ${price} ${status}
    • `; }
